บทที่ 7 : PHP Constants

ค่าคงที่ในคลาส (PHP Constants)

การกำหนดตัวแปรค่าคงที่ด้วยคำสั่ง const และวิธีการเรียกใช้งานผ่านเครื่องหมาย :: และคีย์เวิร์ด self::

Class Constants คืออะไร?

ค่าคงที่ (Constant) คือข้อมูลที่เมื่อประกาศและกำหนดค่าแล้ว จะไม่มีวันเปลี่ยนแปลงได้อีกตลอดการทำงานของโปรแกรม เหมาะสำหรับเก็บค่าคงที่ทางคณิตศาสตร์, รหัสสถานะ (Status Codes) หรือข้อความระบบ

ในคลาสเราใช้คีย์เวิร์ด const (โดยเป็นธรรมเนียมปฏิบัติที่จะตั้งชื่อเป็น ตัวพิมพ์ใหญ่ทั้งหมด) และมีข้อสังเกตสำคัญคือ เวลาประกาศไม่ต้องมีเครื่องหมาย $ นำหน้า

ตัวอย่างการใช้งาน const และ self::

PHP 8+
constants.php 
<?php
class Goodbye {
  // ประกาศค่าคงที่ด้วย const (ตัวพิมพ์ใหญ่ ไม่มี $ หน้าชื่อ)
  const LEAVE_MESSAGE = "ขอบคุณที่เยี่ยมชมเว็บไซต์ของเรา! พบกันใหม่โอกาสหน้าครับ";

  public function byebye() {
    // การเข้าถึงจาก "ภายในคลาสเดียวกัน" ต้องใช้คีย์เวิร์ด self:: ตามด้วยชื่อค่าคงที่
    echo self::LEAVE_MESSAGE;
  }
}

// 1. การเรียกใช้จาก "ภายนอกคลาส" โดยไม่ต้องสร้าง Object (ใช้ ชื่อคลาส::ชื่อค่าคงที่)
echo Goodbye::LEAVE_MESSAGE;
echo "<br><hr>";

// 2. การเรียกใช้ผ่านเมธอดของ Object
$goodbye = new Goodbye();
$goodbye->byebye();
?>
