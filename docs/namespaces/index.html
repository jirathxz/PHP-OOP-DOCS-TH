บทที่ 13 : PHP Namespaces

เนมสเปซ (PHP Namespaces)

การจัดระเบียบคลาสเป็นหมวดหมู่ เปรียบเสมือนโฟลเดอร์เก็บไฟล์ เพื่อป้องกันปัญหาชื่อคลาสหรือฟังก์ชันซ้ำกันในระบบใหญ่

Namespace คืออะไร? มีประโยชน์อย่างไร?

เมื่อโปรเจกต์ของคุณมีขนาดใหญ่ขึ้น หรือมีการดึงไลบรารีของคนอื่นมาใช้ (เช่น ผ่าน Composer) ปัญหาที่มักพบคือ "ชื่อคลาสซ้ำกัน (Name Collision)" เช่น เรามีคลาสชื่อ Table (ตาราง HTML) และไลบรารีฐานข้อมูลก็มีคลาสชื่อ Table (ตารางใน Database) เหมือนกัน!

Namespace ทำหน้าที่เหมือนกับ "โฟลเดอร์" ที่คอยแบ่งแยกคลาสที่มีชื่อเหมือนกันให้อยู่คนละหมวดหมู่ โดยคำสั่ง namespace จะต้องเขียนไว้ที่ บรรทัดแรกสุดของไฟล์ PHP เสมอ

ตัวอย่างการสร้างและการใช้งาน Namespace

PHP 8+
namespaces.php 
<?php
// 1. ประกาศ Namespace ไว้บรรทัดแรกสุดของไฟล์
namespace Html;

class Table {
  public $title = "";
  public $numRows = 0;

  public function message() {
    echo "📊 ตาราง HTML: '{$this->title}' มีจำนวน {$this->numRows} แถว<br>";
  }
}

// =========================================================
// ในอีกไฟล์หนึ่ง หรือเมื่อต้องการเรียกใช้คลาสจาก Namespace อื่น:

// วิธีที่ 1: เรียกชื่อเต็ม (Fully Qualified Name)
$table1 = new \Html\Table();
$table1->title = "รายงานยอดขาย";
$table1->numRows = 5;
$table1->message();

// วิธีที่ 2: ใช้คำสั่ง use ด้านบนไฟล์ เพื่อย่อชื่อหรือตั้งฉายา (Alias)
use Html\Table as HtmlTable;

$table2 = new HtmlTable();
$table2->title = "รายชื่อลูกค้า";
$table2->numRows = 10;
$table2->message();
?>
