บทที่ 3 : PHP Constructor

ตัวสร้างออบเจกต์ (PHP Constructor)

เมธอดพิเศษ __construct() ที่ช่วยลดบรรทัดโค้ดและทำงานอัตโนมัติทันทีที่มีการสร้าง Object ใหม่

Constructor คืออะไร?

ใน PHP เมธอด Constructor จะขึ้นต้นด้วยเครื่องหมายขีดล่างสองตัว (Double Underscore) เสมอ คือ __construct()

เมื่อคุณประกาศฟังก์ชันนี้ไว้ในคลาส PHP จะทำการเรียกใช้เมธอดนี้โดยอัตโนมัติทันทีที่คุณสร้าง Object ใหม่ด้วยคำสั่ง new ทำให้เราสามารถส่งค่าพารามิเตอร์ไปตั้งค่าเริ่มต้นให้กับ Properties ได้ในบรรทัดเดียว ไม่ต้องเสียเวลาเรียกเมธอด set_name() หลายๆ รอบ

ตัวอย่างการใช้ __construct()

PHP 8+
constructor.php 
<?php
class Fruit {
  public $name;
  public $color;

  // เมธอด Constructor จะรับค่าและตั้งค่าเริ่มต้นทันทีตอนสร้าง Object
  public function __construct($name, $color) {
    $this->name = $name;
    $this->color = $color;
  }

  public function get_info() {
    return "ผลไม้: {$this->name} สี: {$this->color}";
  }
}

// สร้าง Object พร้อมส่งค่าเข้าไปใน Constructor ได้เลย (สะดวกและสั้นลงมาก!)
$apple = new Fruit("Apple", "Red");
$banana = new Fruit("Banana", "Yellow");

echo $apple->get_info();  // ผลลัพธ์: ผลไม้: Apple สี: Red
echo "<br>";
echo $banana->get_info(); // ผลลัพธ์: ผลไม้: Banana สี: Yellow
?>

ฟีเจอร์ใหม่ PHP 8: Constructor Property Promotion

ใน PHP 8 ขึ้นไป คุณสามารถย่อโค้ด Constructor ให้สั้นลงไปอีกโดยประกาศ Access Modifier (เช่น public, private) ไว้ในวงเล็บของพารามิเตอร์ได้เลย โดยไม่ต้องประกาศตัวแปรด้านบนคลาส:

public function __construct(public $name, public $color) {} // เขียนแค่นี้เสร็จเลย!
